# Map nested list

        # map nested list
		# LIBRARIES
		suppressPackageStartupMessages({
		library(dplyr)
		library(purrr)
		library(stringr)
		})

		# FUNCTIONS

		mk_secs <- function(x) x[1]*3600 + x[2] * 60 + x[3] * 1

		# DATA

		first_time <- rep("10:10:00",10)
		second_time <- rep("10:11:00",10)
		the_times <- c(first_time,second_time)

		# MAIN

		str_split(the_times,":") %>%
		map(., as.numeric) %>%
		map(., mk_secs) %>%
		unlist(.)
