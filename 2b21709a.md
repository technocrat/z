# geom_function to ggplot arbitrary plots

    library(ggplot2)

    ggplot() + 
      xlim(0,4) + 
      xlab("x") +
      geom_function(fun = function(x) 1*x) +
      geom_function(fun = function(x) 2*x) +
      geom_function(fun = function(x) 3*x) +
      geom_function(fun = function(x) 4*x)
